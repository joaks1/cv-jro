\usepackage{pgf}
\newcommand{\longcvitem}[4][0.16]{
    \pgfmathsetmacro\result{1.0-#1}
    \begin{tabular}{ @{}P{\result} E{#1}@{} }
         #3 & #2 \\
         \multicolumn{2}{P{1.0}}{#4}
    \end{tabular}}

\newcommand{\bflongcvitem}[4][0.16]{
    \pgfmathsetmacro\result{1.0-#1}
    \begin{tabular}{ @{}P{\result} E{#1}@{} }
        \textbf{#3} & \textbf{#2} \\
         \multicolumn{2}{P{1.0}}{#4}
    \end{tabular}}

\newcommand{\revcvitem}[3][0.16]{
    \pgfmathsetmacro\result{1.0-#1}
    \begin{tabular}{ @{}P{\result} E{#1}@{} }
        #3 & #2 \\
    \end{tabular}}

\newcommand{\bfrevcvitem}[3][0.16]{
    \pgfmathsetmacro\result{1.0-#1}
    \begin{tabular}{ @{}P{\result} E{#1}@{} }
        {\bf #3} & \textbf{#2} \\
    \end{tabular}}
